Detailed Description

a QStringListModel, which allows to check/uncheck each entry

Usage in QListView:

Basic usage is simple:

JKQTECheckableStringListModel* model=new JKQTECheckableStringListModel();
// add lines
QStringList lst;
lst<<tr("item 1");
lst<<tr("item 2");
lst<<tr("item 3");
lst<<tr("item 4");
lst<<tr("item 5");
lst<<tr("item 6");
lst<<tr("item 7");
lst<<tr("item 8");
lst<<tr("item 9");
lst<<tr("item 10");
model->setStringList(lst);
// set checkable
model->setCheckable(true);
// set check state
model->setChecked(0);
model->setChecked(1);
model->setChecked(2);
model->setChecked(5);
model->setChecked(8);
model->setChecked(9);
// assign model to view
QListView* wid=new QListView(this);
wid->setModel(model);
